class Solution {
    public int divisorSubstrings(int num, int k) {

        int mod=(int)Math.pow(10,k),m=num;
        int limit=(int)Math.pow(10,k-1)-1,ans=0;
        while(num>limit){
            int n=num%mod;
            if(n!=0&&m%n==0){
                ans++;
            }
            num/=10;
        }
        return ans;
    }
}
